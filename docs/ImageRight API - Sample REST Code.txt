using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

using System.Net;
using Newtonsoft.Json;
using Newtonsoft.Json.Linq;
using System.Net.Http;
using System.Net.Http.Headers;


namespace ImageRightRESTClient
{
    class Program
    {
        private const string BASE_URL = @"https://wsol-8000001app.worksmartonline.vertafore.com/8000001/api/";  //replace with your URL
        private const string AUTHENTICATE_URL = @"/authenticate";
        private const string ATTRIBUTE_DEFINITIONS_URL = @"/attributes/definitions";

        static void Main(string[] args)
        {

            //*********** Authenticate to REST via endpoint: "/authenticate" ***********
            AuthenticateRequestModel authenticateRequestModel = new AuthenticateRequestModel();
            authenticateRequestModel.UserName = "irtestapi";
            authenticateRequestModel.Password = "password"; //replace with actual password

            using (var client = new HttpClient())
            {
                client.BaseAddress = new Uri(BASE_URL);
                client.DefaultRequestHeaders.Accept.Add(new MediaTypeWithQualityHeaderValue("application/json"));

                string fullAddress = BASE_URL + AUTHENTICATE_URL;

                var response = client.PostAsJsonAsync(fullAddress, authenticateRequestModel).Result;
                string authToken = string.Empty;

                if (response.IsSuccessStatusCode)
                {
                    string rawResult = response.Content.ReadAsStringAsync().Result;
                    authToken = rawResult;

		    //*********** Example REST call to endpoint: "/attributes/definitions" ***********
                    client.DefaultRequestHeaders.Authorization = new AuthenticationHeaderValue("AccessToken", authToken);		    

                    var response2 = client.GetAsync(BASE_URL + ATTRIBUTE_DEFINITIONS_URL);
                    string rawResult2 = response2.Result.Content.ReadAsStringAsync().Result;
                }
                else
                {
                    //Console.WriteLine(string.Format("Authentication Error: Status: {0} and returned {1} \r\n", response.StatusCode, response.Content.ReadAsStringAsync().Result));
                }
            }
        }

        private class AuthenticateRequestModel
        {
            public string UserName { get; set; }
            public string Password { get; set; }
        }
    }
}
